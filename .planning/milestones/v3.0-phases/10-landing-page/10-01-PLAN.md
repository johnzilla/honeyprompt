**JavaScript (inline, at end of body, per D-08, D-09, D-10, D-11):**
```javascript
(async function() {
  const output = document.getElementById('stats-output');
  try {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 8000);
    const res = await fetch('https://honeyprompt.sh/stats', { signal: controller.signal });
    clearTimeout(timeout);
    if (!res.ok) throw new Error('HTTP ' + res.status);
    const d = await res.json();
    output.innerHTML =
      '  sessions: <span class="val">' + d.total_sessions + '</span>' +
      '  |  url_fetch: <span class="val">' + d.tier1_sessions + '</span>' +
      '  |  conditional: <span class="val">' + d.tier2_sessions + '</span>' +
      '  |  composed: <span class="val">' + d.tier3_sessions + '</span>';
  } catch (e) {
    output.innerHTML = '  <span class="err">Stats unavailable. Honeypot may be offline.</span> ' +
      '<a href="https://github.com/johnzilla/honeyprompt">View on GitHub</a>';
  }
})();
```
