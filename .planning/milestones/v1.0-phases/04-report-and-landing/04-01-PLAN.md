```rust
fn classify_label(classification: &AgentClass) -> String {
    match classification {
        AgentClass::KnownCrawler { provider } => format!("KnownCrawler:{}", provider),
        AgentClass::KnownAgent { provider } => format!("KnownAgent:{}", provider),
        AgentClass::Unknown => "Unknown".to_string(),
```

    ```rust
    Commands::Report(args) => {
        let path = &args.path;
        let db_path = path.join(".honeyprompt").join("events.db");
        let conn = store::open_or_create_db(&db_path)?;
        let markdown = report::generate_report(&conn)?;
        if args.stdout {
            print!("{}", markdown);
        } else {
            let out_path = if let Some(ref p) = args.output {
                p.clone()
            } else {
                path.join("report.md")
            };
            std::fs::write(&out_path, &markdown)?;
            println!("Report written to {}", out_path.display());
        }
        Ok(())
    }
    ```

    Note: The default output path is `report.md` in the project directory (per D-05). The --output flag overrides this. The --stdout flag prints to stdout instead (per D-06).
